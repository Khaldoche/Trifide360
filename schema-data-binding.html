6. Data Binding & MVVM - Liaison Données-Interface

Vue d'Ensemble du Système Data Binding

Le système Data Binding de Trifide 360 implémente le pattern MVVM pour séparer la logique métier de l'interface utilisateur. Il offre une liaison bidirectionnelle entre les données géospatiales et les contrôles WPF, avec notifications automatiques et synchronisation en temps réel.

ICommand

// ViewModel pour gestion couche cartographique public class LayerViewModel : ViewModelBase { private LayerData _layerData; private bool _isVisible = true; private double _transparency = 1.0; public LayerData LayerData { get => _layerData; set => SetProperty(ref _layerData, value); } public bool IsVisible { get => _isVisible; set { if (SetProperty(ref _isVisible, value)) { UpdateLayerVisibility(); } } } public double Transparency { get => _transparency; set => SetProperty(ref _transparency, Math.Max(0, Math.Min(1, value))); } public ICommand ToggleVisibilityCommand { get; } public ICommand RemoveLayerCommand { get; } public LayerViewModel(LayerData layerData) { _layerData = layerData; ToggleVisibilityCommand = new RelayCommand(ToggleVisibility); RemoveLayerCommand = new RelayCommand(RemoveLayer, CanRemoveLayer); } }

Collection Observable

// Collection observable pour features public class FeatureCollectionViewModel : ViewModelBase { private ObservableCollection<FeatureViewModel> _features; private FeatureViewModel _selectedFeature; public ObservableCollection<FeatureViewModel> Features { get => _features; set => SetProperty(ref _features, value); } public FeatureViewModel SelectedFeature { get => _selectedFeature; set => SetProperty(ref _selectedFeature, value); } // Commandes public ICommand AddFeatureCommand { get; } public ICommand DeleteSelectedCommand { get; } public ICommand RefreshCommand { get; } public FeatureCollectionViewModel() { _features = new ObservableCollection<FeatureViewModel>(); AddFeatureCommand = new RelayCommand(AddNewFeature); DeleteSelectedCommand = new RelayCommand(DeleteSelected, () => SelectedFeature != null); RefreshCommand = new AsyncRelayCommand(RefreshFeaturesAsync); } }
